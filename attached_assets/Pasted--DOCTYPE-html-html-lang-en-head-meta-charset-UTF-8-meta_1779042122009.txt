function toggleOpt(qid,oid,isMulti,max){

function goNext(){
  var q=questions[current];
  if(!answers[q.id]||answers[q.id].length===0)return;
  if(q.type==="single")answers[q.id]=answers[q.id][0];
  if(current===questions.length-1){renderResult();return;}
  current++;renderQuestion(current);
}

function goBack(){

}
